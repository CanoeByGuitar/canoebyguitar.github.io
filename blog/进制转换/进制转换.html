进制转换

June 22, 2020 · 2 min read

PAT中常用的进制转化，包括string、vector和int间的按进制转换。string类型可以直接使用iostream读入:cin >> str 具体写法如下

#include <iostream>
#include <vector>
#include <string>
#include <cmath>

using namespace std;

int toNumber_str(string x, int radix) {
    int a = 0;
    for (int i = 0; i < x.size(); i++) {
        a = a * radix + (x[i] - '0');
    }
    return a;
}

int toNumber_vec(vector<int> x, int radix) {
    int a = 0;
    for (int i = 0; i < x.size(); i++) {
        a = a * radix + x[i];
    }
    return a;
}

string toString(int num,int radix){
    string s = "";
    while(num!=0){
        char a[10];
        sprintf(a,"%d",num % radix);
        string str = a;
        s += a;
        num /= radix;
    }
    return s;
}

vector<int> toVector(int num,int radix){
    vector<int> vec;
    while(num!=0){
        vec.push_back(num % radix);
        num /= radix;
    }
    return vec;
}

int main() {
    /*
     * 字符串,int数组按进制转int
     */
    string s = "12345";
    vector<int> vec;
    for(int i = 1;i <= 5;i++){
        vec.push_back(i);
    }
    cout << toNumber_str(s,10)<<endl;//12345
    cout << toNumber_vec(vec,10)<<endl;//12345

    /*
     * int按进制转字符串,int数组（逆向）
     */
    int a = 12345;
    vector<int> v = toVector(a,10);//"54321"
    string str = toString(a,10);//"54321"

    /*
     * int转字符串（正向，仅支持10进制）
     */
    char b[10];
    int size = sprintf(b,"%d",a);
    for(int i = 0;i < size;i++) cout << b[i];//"12345"
}
